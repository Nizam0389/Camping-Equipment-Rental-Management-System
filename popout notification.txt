1. Basic Popup
<button class="button" data-for="js_basic-popup">Show basic popup</button>
<div class="popup js_basic-popup">
    <div class="popup__background"></div>
    <div class="popup__content">
        <h3 class="popup__content__title">Sample Popup</h3>
        <p>Lorem Ipsum dolor sit amet!</p>
        <button class="button" data-for="js_basic-popup">Hide Popup</button>
    </div>
</div>
const addButtonTrigger = (el) => {
    el.addEventListener('click', () => {
        const popupEl = document.querySelector(`.${el.dataset.for}`);
        popupEl.classList.toggle('popup--visible');
    });
};
document.querySelectorAll('button[data-for="js_basic-popup"]').forEach(addButtonTrigger);

Shared CSS for All Popups
/* General popup styles */
.popup {
    visibility: hidden;
    position: fixed;
    top: 0;
    left: 0;
    width: 100%;
    height: 100%;
    display: flex;
    align-items: center;
    justify-content: center;
    z-index: 10000;
    transition: visibility 0s linear 0.3s;
}

.popup--visible {
    visibility: visible;
    transition-delay: 0s;
}

.popup__background {
    position: absolute;
    width: 100%;
    height: 100%;
    background: rgba(0, 0, 0, 0.4);
    transition: opacity 0.3s ease-in-out;
}

.popup--visible .popup__background {
    opacity: 1;
}

    position: relative;

    opacity: 0;

}

    opacity: 1;

}
